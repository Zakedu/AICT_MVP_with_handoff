CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email           VARCHAR(255) UNIQUE NOT NULL,
    password_hash   VARCHAR(255) NOT NULL,
    name            VARCHAR(100) NOT NULL,
    role            VARCHAR(20) DEFAULT 'examinee',  -- examinee, admin, grader
    organization    VARCHAR(255),
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

-- μ‹ν— μ„Έμ… ν…μ΄λΈ”
CREATE TABLE exam_sessions (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id),
    status          VARCHAR(20) DEFAULT 'pending',  -- pending, in_progress, submitted, graded
    selected_roles  JSONB,                          -- μ„ νƒν• μ§κµ°
    started_at      TIMESTAMP,
    submitted_at    TIMESTAMP,
    time_limit      INTEGER DEFAULT 3600,           -- μ΄ λ‹¨μ„
    created_at      TIMESTAMP DEFAULT NOW()
);

-- μ¶”μ¶λ λ¬Έν•­ μ„ΈνΈ
CREATE TABLE exam_question_sets (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id      UUID REFERENCES exam_sessions(id),
    part            INTEGER NOT NULL,               -- 1, 2, 3
    questions       JSONB NOT NULL,                 -- μ¶”μ¶λ λ¬Έν•­ (μ…”ν”λ μƒνƒ)
    created_at      TIMESTAMP DEFAULT NOW()
);

-- λ‹µμ• ν…μ΄λΈ”
CREATE TABLE answers (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id      UUID REFERENCES exam_sessions(id),
    question_id     VARCHAR(50) NOT NULL,
    part            INTEGER NOT NULL,
    answer          JSONB,                          -- μ ν•λ³„ λ‹¤λ¥Έ κµ¬μ΅°
    flagged         BOOLEAN DEFAULT FALSE,
    time_spent      INTEGER,                        -- λ¬Έν•­λ³„ μ†μ”μ‹κ°„
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW(),
    UNIQUE(session_id, question_id)
);

-- μ±„μ  κ²°κ³Ό ν…μ΄λΈ”
CREATE TABLE scores (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id      UUID REFERENCES exam_sessions(id),
    question_id     VARCHAR(50) NOT NULL,
    part            INTEGER NOT NULL,
    score           DECIMAL(5,2),
    max_score       DECIMAL(5,2),
    competency      VARCHAR(50),                    -- μ—­λ‰ μμ—­
    auto_graded     BOOLEAN DEFAULT TRUE,
    ai_feedback     TEXT,                           -- AI μƒμ„± ν”Όλ“λ°±
    grader_id       UUID REFERENCES users(id),      -- μλ™ κ²€ν† μ
    reviewed_at     TIMESTAMP,
    created_at      TIMESTAMP DEFAULT NOW()
);

-- λ¬Έν•­ ν…μ΄λΈ” (Master)
CREATE TABLE questions (
    id              VARCHAR(50) PRIMARY KEY,
    part            INTEGER NOT NULL,
    type            VARCHAR(30) NOT NULL,
    competency      VARCHAR(50) NOT NULL,
    content         JSONB NOT NULL,                 -- λ¬Έν•­ λ‚΄μ© μ „μ²΄
    difficulty      INTEGER DEFAULT 3,              -- 1~5
    active          BOOLEAN DEFAULT TRUE,
    created_by      UUID REFERENCES users(id),
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

-- μ±„μ  κΈ°μ¤€ ν…μ΄λΈ”
CREATE TABLE rubrics (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    question_id     VARCHAR(50) REFERENCES questions(id),
    weight          DECIMAL(3,2) DEFAULT 1.0,
    total_points    INTEGER DEFAULT 100,
    pass_threshold  INTEGER DEFAULT 80,
    criteria        JSONB NOT NULL,                 -- λ λ²¨λ³„ κΈ°μ¤€
    created_at      TIMESTAMP DEFAULT NOW()
);

-- λ΅κ·Έ ν…μ΄λΈ” (ν–‰λ™ μ¶”μ )
CREATE TABLE activity_logs (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id      UUID REFERENCES exam_sessions(id),
    user_id         UUID REFERENCES users(id),
    action          VARCHAR(50) NOT NULL,           -- page_view, answer_submit, flag_toggle, etc.
    metadata        JSONB,
    ip_address      INET,
    user_agent      TEXT,
    created_at      TIMESTAMP DEFAULT NOW()
);

-- μΈλ±μ¤
CREATE INDEX idx_sessions_user ON exam_sessions(user_id);
CREATE INDEX idx_sessions_status ON exam_sessions(status);
CREATE INDEX idx_answers_session ON answers(session_id);
CREATE INDEX idx_scores_session ON scores(session_id);
CREATE INDEX idx_questions_part ON questions(part, competency);
CREATE INDEX idx_logs_session ON activity_logs(session_id);
